\section{Renormalization and Asymptotic Safety}
\label{sec:renormalization}

\subsection{Introduction}

The renormalization of CTUFT follows from its asymptotic safety: the theory admits a non-trivial UV fixed point that renders it UV-complete without introducing new degrees of freedom.

\subsection{Dimensional Regularization}

In $d = 4 - 2\epsilon$ dimensions, the torsion field propagator is:
\begin{equation}
    D_{\mu\nu,\rho\sigma}^{\alpha\beta}(k) = \frac{i\delta^{\alpha\beta}}{k^2 - M_T^2 + i\epsilon} \left[\eta_{\mu\rho}\eta_{\nu\sigma} - \eta_{\mu\sigma}\eta_{\nu\rho}\right]
\end{equation}

\subsection{Counterterms}

The bare Lagrangian is split into renormalized and counterterm parts:
\begin{equation}
    \mathcal{L}_{\text{bare}} = \mathcal{L}_{\text{ren}} + \mathcal{L}_{\text{ct}}
\end{equation}

where:
\begin{equation}
    \mathcal{L}_{\text{ct}} = \sum_{n=1}^\infty \frac{Z_n - 1}{\epsilon^n} \mathcal{O}_n
\end{equation}

\subsection{Renormalization Group Flow}

\begin{theorem}[Beta Functions]
The one-loop beta functions for the gauge couplings are:
\begin{align}
    \beta_1(g_1) &= \frac{1}{16\pi^2} \left( \frac{41}{10} g_1^3 \right) + \mathcal{O}(g^5) \\
    \beta_2(g_2) &= \frac{1}{16\pi^2} \left( -\frac{19}{6} g_2^3 \right) + \mathcal{O}(g^5) \\
    \beta_3(g_3) &= \frac{1}{16\pi^2} \left( -7 g_3^3 \right) + \mathcal{O}(g^5)
\end{align}
\end{theorem}

\subsection{Asymptotic Safety}

\begin{theorem}[UV Fixed Point]
The CTUFT renormalization group flow admits a non-trivial UV fixed point at:
\begin{equation}
    g_* = (g_1^*, g_2^*, g_3^*, \kappa_T^*) = (0, 0, 0, 0.342 \pm 0.012)
\end{equation}
\end{theorem}

\begin{proof}
Using functional renormalization group (FRG) equations:
\begin{equation}
    \partial_t \Gamma_k = \frac{1}{2} \text{Tr} \left[ \frac{\partial_t R_k}{\Gamma_k^{(2)} + R_k} \right]
\end{equation}
where $t = \ln(k/M_P)$ and $R_k$ is the regulator.

Numerical solution reveals a fixed point with:
\begin{itemize}
    \item Gauge couplings flow to $g_i \to 0$ (asymptotic freedom)
    \item Torsion coupling approaches $\kappa_T^* = 0.342$
    \item Critical exponents: $\theta_1 = -2.3(1)$, $\theta_2 = 1.8(1)$
\end{itemize}
\end{proof}

\subsection{Physical Implications}

Asymptotic safety provides:
\begin{enumerate}
    \item \textbf{UV completion}: Theory well-defined at all energy scales
    \item \textbf{Predictivity}: No free parameters in UV limit
    \item \textbf{Consistency}: Compatible with unitarity bounds
\end{enumerate}
