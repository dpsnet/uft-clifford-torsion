\begin{align}

\begin{align}
    \gamma_{1st} &\approx 0.03 \\
    \gamma_{2nd} &\approx 0.06 \\

\begin{equation}
    \frac{m_t}{m_c} \sim \left(\frac{M_P}{E_{EW}}\right)^{0.03} \sim 10^{2.5}, \quad

consistent with observed values $m_t/m_c \approx 10^{2.7}$ and $m_c/m_u \approx 10^{2.4}$.

\begin{equation}
