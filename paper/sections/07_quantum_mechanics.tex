\end{insight}

\begin{equation}
    |\psi_{AB}\rangle = \frac{1}{\sqrt{2}}(|0_A 1_B\rangle - |1_A 0_B\rangle)
\end{equation}

\begin{equation}
    t_{\text{int}}^A = t_{\text{int}}^B = t_{\text{int}}^{\text{(shared)}}
    \quad \text{while} \quad
    t_{\text{rec}}^A \neq t_{\text{rec}}^B
\end{equation}
