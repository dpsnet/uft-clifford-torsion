\section{从10D几何导出完整QFT拉格朗日量}
\label{sec:qft_lagrangian_complete}

\subsection{引言}

本节介绍从10维内空间几何严格推导完整标准模型（加引力）拉格朗日量。从克利福德代数 $Cl(1,9)$ 出发，我们通过维度约化推导拉格朗日量中的每一项。

\subsection{10D中的主拉格朗日量}

\subsubsection{几何作用量}

10维中的基本作用量为：
\begin{equation}
    S_{10D} = \int d^{10}x \sqrt{-g_{10}} \left[ \frac{1}{2\kappa_{10}^2}R_{10} - \frac{1}{4}F_{MN}F^{MN} + i\bar{\Psi}\Gamma^M D_M \Psi \right]
\end{equation}

其中：
\begin{itemize}
    \item $R_{10}$ 是10D里奇标量
    \item $F_{MN}$ 是规范场的场强
    \item $\Psi$ 是10D马约拉纳-外尔旋量
    \item $\Gamma^M$ 是 $Cl(1,9)$ 伽马矩阵
\end{itemize}

\subsubsection{克利福德代数 $Cl(1,9)$}

伽马矩阵满足：
\begin{equation}
    \{\Gamma^M, \Gamma^N\} = 2\eta^{MN}, \quad M,N = 0,1,\ldots,9
\end{equation}

度规矩迹为 $\eta = \text{diag}(-,+,+,+,+,+,+,+,+,+)$。

10维旋量表示是32维的（马约拉纳-外尔）。

\subsection{维度约化：10D → 4D}

\subsubsection{卡鲁扎-克莱因拟设}

我们将10维坐标分解为 $x^M = (x^\mu, y^a)$，其中：
\begin{itemize}
    \item $x^\mu$：4维互反空间（$\mu = 0,1,2,3$）
    \item $y^a$：6维内方向（$a = 4,\ldots,9$）
\end{itemize}

度规拟设为：
\begin{equation}
    ds_{10}^2 = e^{-\Phi}g_{\mu\nu}dx^\mu dx^\nu + g_{ab}(dy^a + A^a_\mu dx^\mu)(dy^b + A^b_\nu dx^\nu)
\end{equation}

其中：
\begin{itemize}
    \item $g_{\mu\nu}$ 是4维度规
    \item $g_{ab}$ 是内部度规
    \item $A^a_\mu$ 是规范场（卡鲁扎-克莱因矢量）
    \item $\Phi$ 是伸缩子
\end{itemize}

\subsubsection{规范对称性的涌现}

内空间的等距变换成为4维中的规范对称性：
\begin{equation}
    \text{Isom}(M_{\text{int}}) = U(1) \times SU(2) \times SU(3)
\end{equation}

\subsection{引力部分}

\subsubsection{爱因斯坦-希尔伯特项}

10D里奇标量的约化给出：
\begin{equation}
    \frac{1}{2\kappa_{10}^2}\int d^{10}x \sqrt{-g_{10}}R_{10} = \frac{1}{2\kappa_4^2}\int d^4x \sqrt{-g_4}\left(R_4 - \frac{1}{2}\partial_\mu\Phi\partial^\mu\Phi + \cdots\right)
\end{equation}

其中4维引力常数为：
\begin{equation}
    \frac{1}{\kappa_4^2} = \frac{V_{\text{int}}}{\kappa_{10}^2} = \frac{M_P^2}{16\pi}
\end{equation}

\subsubsection{挠率项}

CTUFT修正表现为：
\begin{equation}
    \mathcal{L}_{\text{torsion}} = \frac{1}{2\kappa_4^2}\left(2\Lambda + \tau_0^2 R + \tau_0^4 R_{\mu\nu}R^{\mu\nu} + \cdots\right)
\end{equation}

其中 $\Lambda = 3\tau_0^2/2\kappa_4^2$ 是宇宙学常数。

\subsection{规范部分}

\subsubsection{$U(1)_Y$：超荷规范场}

来自10D规范场 $A_M$：
\begin{equation}
    \mathcal{L}_{U(1)} = -\frac{1}{4g_Y^2}F_{\mu\nu}F^{\mu\nu}, \quad F_{\mu\nu} = \partial_\mu A_\nu - \partial_\nu A_\mu
\end{equation}

规范耦合由内空间几何确定：
\begin{equation}
    g_Y = \sqrt{\frac{3}{5}}\tau_0 \cdot g_{\text{GUT}}
\end{equation}

\subsubsection{$SU(2)_L$：弱规范场}

$SU(2)$ 规范场来自 $S^2$ 因子的等距群：
\begin{equation}
    \mathcal{L}_{SU(2)} = -\frac{1}{2g_2^2}\text{Tr}(W_{\mu\nu}W^{\mu\nu})
\end{equation}

其中：
\begin{equation}
    W_{\mu\nu}^a = \partial_\mu W_\nu^a - \partial_\nu W_\mu^a + g_2\epsilon^{abc}W_\mu^b W_\nu^c
\end{equation}

\subsubsection{$SU(3)_C$：色规范场}

$SU(3)$ 规范场来自 $CP^2$ 或卡拉比-丘因子的等距：
\begin{equation}
    \mathcal{L}_{SU(3)} = -\frac{1}{2g_3^2}\text{Tr}(G_{\mu\nu}G^{\mu\nu})
\end{equation}

\subsection{费米子部分}

\subsubsection{狄拉克拉格朗日量}

10D狄拉克项的约化：
\begin{equation}
    i\bar{\Psi}\Gamma^M D_M \Psi \to i\bar{\psi}\gamma^\mu D_\mu \psi + \text{汤川耦合项}
\end{equation}

4维旋量通过分解32分量10D旋量得到：
\begin{equation}
    \mathbf{32}_{SO(1,9)} \to (\mathbf{2}, \mathbf{1}, \mathbf{3}) \oplus (\mathbf{2}, \mathbf{1}, \bar{\mathbf{3}}) \oplus (\mathbf{1}, \mathbf{2}, \mathbf{3}) \oplus \cdots
\end{equation}

\subsubsection{规范相互作用}

协变导数为：
\begin{equation}
    D_\mu = \partial_\mu - ig_Y Y B_\mu - ig_2 T^a W_\mu^a - ig_3 T^A G_\mu^A
\end{equation}

\subsubsection{手征结构}

10D马约拉纳-外尔条件投影为4D中的手征费米子：
\begin{align}
    \psi_L &: \quad \text{左手二重态} \\
    \psi_R &: \quad \text{右手单态}
\end{align}

这是弱相互作用中宇称破坏的几何起源。

\subsection{希格斯部分}

\subsubsection{作为内空间模式的希格斯场}

希格斯场对应于内空间度规的涨落：
\begin{equation}
    \phi(x) = \delta g_{ab}(x) \cdot \epsilon^{ab}
\end{equation}

\subsubsection{希格斯势}

势能从内空间的曲率产生：
\begin{equation}
    V(\phi) = -\mu^2|\phi|^2 + \lambda|\phi|^4
\end{equation}

其中：
\begin{align}
    \mu^2 &= \frac{\tau_0^2}{V_{\text{int}}} \sim (100 \text{ GeV})^2 \\
    \lambda &= \frac{\tau_0^4}{8\pi^2} \sim 0.13
\end{align}

\subsubsection{希格斯VEV和质量}

真空期望值：
\begin{equation}
    v = \sqrt{\frac{\mu^2}{\lambda}} = \frac{2\pi}{\tau_0^2}\sqrt{V_{\text{int}}} \approx 246 \text{ GeV}
\end{equation}

希格斯质量：
\begin{equation}
    m_H = \sqrt{2\lambda}v \approx 125 \text{ GeV}
\end{equation}

\subsection{汤川耦合}

\subsubsection{几何起源}

汤川耦合是重叠积分：
\begin{equation}
    Y_{ij} = \tau_0 \int_{\text{int}} d^6y \sqrt{g_{\text{int}}} \, \psi_i^\dagger(y) \psi_j(y) \phi(y)
\end{equation}

\subsubsection{质量产生}

电弱对称性破缺后：
\begin{equation}
    m_{ij} = Y_{ij} \cdot v
\end{equation}
