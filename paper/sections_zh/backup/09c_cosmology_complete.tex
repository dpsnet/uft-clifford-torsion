\subsection{引言}

UFT中的宇宙学不仅是将理论应用于宇宙——它是对10D→4D过渡的基本检验。早期宇宙提供了唯一接近临界能标$E_c$的能量的实验室，使宇宙学观测成为UFT的关键检验。

\subsection{来自谱维过渡的暴涨}

\subsubsection{标准暴涨问题}

传统暴涨需要：
\begin{itemize}
    \item 具有非常平坦势的暴胀子场
    \item 精细调节的初始条件
    \item 结束暴涨并再热宇宙的机制
\end{itemize}

\subsubsection{UFT解决：几何暴涨}

\begin{insight}[暴涨作为10D→4D过渡]
在UFT中，暴涨不是由标量场驱动，而是由宇宙冷却时从10D到4D的几何过渡驱动。有效维数变化为：
\begin{equation}
    d_s(T) = 4 + \frac{6}{1+(T/T_c)^2}
\end{equation}
其中$T_c \sim E_c$是临界温度。
\end{insight}

\subsubsection{E-折叠数}

\begin{theorem}[来自维数变化的E-折叠]
暴涨的e-折叠数由维数过渡的速率决定：
\begin{equation}
    N_e = \int_{t_i}^{t_f} H \, dt = \frac{1}{2}\ln\left(\frac{d_s(t_i)-4}{d_s(t_f)-4}\right) \approx 60
\end{equation}
\end{theorem}

\begin{proof}
过渡期间，哈勃参数随有效维数缩放：
\begin{equation}
    H \sim \frac{1}{d_s-4} \frac{d}{dt}(d_s-4)
\end{equation}

从$d_s \approx 10$到$d_s \approx 4 + \epsilon$积分：
\begin{equation}
    N_e = \int_{10}^{4+\epsilon} \frac{d(d_s)}{d_s-4} = \ln\left(\frac{6}{\epsilon}\right)
\end{equation}

对于$\epsilon \sim 10^{-26}$（对应观测曲率），我们得到$N_e \approx 60$。
\end{proof}

\subsubsection{原初功率谱}

标量功率谱被维数流修正：
\begin{equation}
    P_s(k) = P_s^{(0)}(k) \cdot \left[1 + \tau_0^2 \cdot \ln(k/k_*)\right]
\end{equation}

这预言谱指数的小跑动：
\begin{equation}
    \alpha_s = \frac{d n_s}{d \ln k} = -2\tau_0^2 \sim -5 \times 10^{-5}
\end{equation}

与Planck约束一致（$\alpha_s = -0.0045 \pm 0.0067$）。

\subsection{原初黑洞形成}

\subsubsection{机制：扭转场涨落}

\begin{insight}[来自密度峰的PBH]
在10D→4D过渡期间，扭转场有涨落$\delta\tau/\tau \sim \mathcal{O}(1)$。$\delta\tau \u003e \tau_c$的区域坍缩形成原初黑洞。
\end{insight}

\begin{theorem}[PBH质量函数]
原初黑洞的质量谱为：
\begin{equation}
    \frac{dn}{dM} = \beta \cdot \left(\frac{M}{M_*}\right)^{-\alpha} \cdot \exp\left[-\left(\frac{M}{M_*}\right)^2\right]
\end{equation}
其中：
\begin{itemize}
    \item $M_* = M_{\text{Planck}} \cdot (E_c/E_{\text{form}})^2 \cdot \tau_0^{3/2} \sim 10^{-12} M_\odot$
    \item $\alpha = 1 + \tau_0 \approx 1.005$
    \item $\beta \sim 10^{-8}$（形成比例）
\end{itemize}
\end{theorem}

\subsubsection{暗物质丰度}

\begin{equation}
    \Omega_{\text{PBH}} = \int_{M_{\text{min}}}^{M_{\text{max}}} \frac{dn}{dM} \cdot \frac{M}{\rho_c} \, dM \approx 0.26
\end{equation}

这自然解释暗物质丰度，无需新粒子。

\subsubsection{重子数破坏}

在UFT中，重子数不是基本对称性，而是仅在4D低能极限有效的涌现守恒律。在$E \sim E_c$，（通过统一规范玻色子）夸克和轻子之间的跃迁破坏$B$和$L$但保持$B-L$。

\subsubsection{重子不对称性公式}

\begin{theorem}[重子-光子比]
观测到的重子不对称性被预测为：
\begin{equation}
    \eta_B = \frac{n_B}{n_\gamma} = \frac{28}{79} \cdot \frac{\Gamma_{\text{CPV}}}{H} \cdot \epsilon_{\text{washout}} \approx 6 \times 10^{-10}
\end{equation}
其中：
\begin{itemize}
    \item $\Gamma_{\text{CPV}}$是CP破坏相互作用率
    \item $H$是过渡期间的哈勃率
    \item $\epsilon_{\text{washout}} = \tau_0^2$是洗涤抑制因子
\end{itemize}
\end{theorem}

\begin{proof}
在10D→4D过渡期间，破坏$B+L$但保持$B-L$的sphaleron过程处于平衡。扭转场中的CP破坏相位创建净重子不对称性：

\begin{equation}
    n_B \sim \sum_f \mu_f \cdot \Gamma_{\text{CPV}} \cdot \tau_{\text{trans}}
\end{equation}

其中$\mu_f$是化学势，$\tau_{\text{trans}} \sim 1/E_c$是过渡时间尺度。

对于$\Gamma_{\text{CPV}} \sim \alpha_{\text{GUT}}^2 T$和$H \sim T^2/M_P$，并计入洗涤：
\begin{equation}
    \eta_B \sim 10^{-2} \cdot \alpha_{\text{GUT}}^2 \cdot \tau_0^2 \sim 6 \times 10^{-10}
\end{equation}
\end{proof}

这与观测值$\eta_B^{(\text{obs})} = (6.12 \pm 0.04) \times 10^{-10}$匹配。
